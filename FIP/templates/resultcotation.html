    def CoinDate(date):

    date_clock = date[11:20:]

    date_ajust = date[:10:]

    date_day = date_ajust[8:10:]

    date_year = date_ajust[:4:]

    date_month_test = date_ajust[5:7:]

    date_date = '{}/{}/{} {}'.format(date_day, date_month_test, date_year, date_clock)

     

    return date_date
